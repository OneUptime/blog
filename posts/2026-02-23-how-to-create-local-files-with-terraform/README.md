# How to Create Local Files with Terraform

Author: [nawazdhandala](https://github.com/nawazdhandala)

Tags: Terraform, Local Provider, File Creation, Infrastructure as Code, Configuration Management

Description: Learn how to create local files with Terraform using the local_file resource for generating configuration files, scripts, templates, and deployment artifacts.

---

Creating local files with Terraform is a common requirement in infrastructure workflows. You might need to generate configuration files for applications, create deployment scripts, produce inventory files for configuration management tools, or output data that other processes consume. The local_file resource in Terraform handles all of these scenarios, creating files on the machine where Terraform runs as part of your infrastructure deployment.

In this guide, we will cover the local_file resource in depth, including content generation, file permissions, template rendering, and integration patterns with other Terraform resources.

## Provider Setup

```hcl
# main.tf
terraform {
  required_version = ">= 1.5.0"
  required_providers {
    local = {
      source  = "hashicorp/local"
      version = "~> 2.5"
    }
    aws = {
      source  = "hashicorp/aws"
      version = "~> 5.0"
    }
  }
}

provider "aws" {
  region = "us-east-1"
}

variable "environment" {
  type    = string
  default = "production"
}
```

## Creating a Simple Text File

```hcl
# simple.tf - Create a basic text file
resource "local_file" "readme" {
  filename = "${path.module}/output/README.txt"
  content  = <<-EOT
    Infrastructure deployed by Terraform
    Environment: ${var.environment}
    Generated at: ${timestamp()}

    Do not edit this file manually.
  EOT
}
```

## Creating JSON Configuration Files

```hcl
# json-config.tf - Generate JSON configuration
resource "local_file" "app_config" {
  filename = "${path.module}/output/config.json"
  content = jsonencode({
    app = {
      name        = "my-application"
      environment = var.environment
      version     = var.app_version
    }
    database = {
      host     = var.db_host
      port     = 5432
      database = "appdb"
      ssl      = true
    }
    cache = {
      host = var.redis_host
      port = 6379
      ttl  = 3600
    }
    features = {
      new_dashboard  = var.environment != "production"
      beta_api       = false
      rate_limiting  = true
    }
  })

  file_permission = "0644"
}

variable "app_version" {
  type    = string
  default = "2.0.0"
}

variable "db_host" {
  type    = string
  default = "db.internal.example.com"
}

variable "redis_host" {
  type    = string
  default = "redis.internal.example.com"
}
```

## Creating YAML Configuration Files

```hcl
# yaml-config.tf - Generate YAML configuration
resource "local_file" "k8s_values" {
  filename = "${path.module}/output/values.yaml"
  content = yamlencode({
    replicaCount = var.environment == "production" ? 3 : 1
    image = {
      repository = "myapp"
      tag        = var.app_version
    }
    service = {
      type = "ClusterIP"
      port = 80
    }
    ingress = {
      enabled = true
      hosts = [{
        host  = "app.${var.domain}"
        paths = [{ path = "/", pathType = "Prefix" }]
      }]
    }
    resources = {
      limits = {
        cpu    = "500m"
        memory = "512Mi"
      }
      requests = {
        cpu    = "250m"
        memory = "256Mi"
      }
    }
  })
}

variable "domain" {
  type    = string
  default = "example.com"
}
```

## Using Templates for File Content

```hcl
# templates.tf - Generate files from templates
resource "local_file" "nginx_config" {
  filename = "${path.module}/output/nginx.conf"
  content = templatefile("${path.module}/templates/nginx.conf.tpl", {
    server_name = "api.${var.domain}"
    upstream_port = 8080
    ssl_enabled = var.environment == "production"
    max_body_size = "10m"
    worker_connections = var.environment == "production" ? 4096 : 1024
  })
  file_permission = "0644"
}

# Generate a docker-compose file
resource "local_file" "docker_compose" {
  filename = "${path.module}/output/docker-compose.yml"
  content = yamlencode({
    version = "3.8"
    services = {
      app = {
        image = "myapp:${var.app_version}"
        ports = ["8080:8080"]
        environment = {
          DB_HOST     = var.db_host
          REDIS_HOST  = var.redis_host
          ENVIRONMENT = var.environment
        }
      }
    }
  })
}
```

## Creating Shell Scripts

```hcl
# scripts.tf - Generate executable scripts
resource "local_file" "deploy_script" {
  filename        = "${path.module}/output/deploy.sh"
  file_permission = "0755"
  content = <<-SCRIPT
    #!/bin/bash
    # Auto-generated deployment script
    # Environment: ${var.environment}

    set -euo pipefail

    echo "Deploying version ${var.app_version} to ${var.environment}..."

    # Pull the latest image
    docker pull myapp:${var.app_version}

    # Update the service
    docker service update --image myapp:${var.app_version} app-service

    echo "Deployment complete."
  SCRIPT
}

resource "local_file" "healthcheck_script" {
  filename        = "${path.module}/output/healthcheck.sh"
  file_permission = "0755"
  content = <<-SCRIPT
    #!/bin/bash
    # Health check script for ${var.environment}

    ENDPOINTS=(
      "https://api.${var.domain}/health"
      "https://app.${var.domain}/health"
    )

    for endpoint in "$${ENDPOINTS[@]}"; do
      status=$(curl -s -o /dev/null -w "%%{http_code}" "$endpoint")
      if [ "$status" != "200" ]; then
        echo "FAILED: $endpoint returned $status"
        exit 1
      fi
      echo "OK: $endpoint"
    done

    echo "All health checks passed."
  SCRIPT
}
```

## Creating Files from Terraform Outputs

```hcl
# output-files.tf - Write Terraform outputs to files
data "aws_caller_identity" "current" {}

resource "local_file" "terraform_outputs" {
  filename = "${path.module}/output/terraform-outputs.json"
  content = jsonencode({
    account_id  = data.aws_caller_identity.current.account_id
    environment = var.environment
    region      = "us-east-1"
    deployed_at = timestamp()
  })
}
```

## Creating Multiple Files with for_each

```hcl
# multi-files.tf - Generate multiple files from a map
variable "env_files" {
  type = map(map(string))
  default = {
    "development" = {
      LOG_LEVEL = "debug"
      DEBUG     = "true"
      API_URL   = "http://localhost:8080"
    }
    "staging" = {
      LOG_LEVEL = "info"
      DEBUG     = "false"
      API_URL   = "https://staging-api.example.com"
    }
    "production" = {
      LOG_LEVEL = "warn"
      DEBUG     = "false"
      API_URL   = "https://api.example.com"
    }
  }
}

resource "local_file" "env" {
  for_each = var.env_files

  filename = "${path.module}/output/envs/${each.key}.env"
  content = join("\n", [
    for key, value in each.value : "${key}=${value}"
  ])
  file_permission = "0644"
}
```

## Controlling File Directory Permissions

```hcl
# directory.tf - Create files in specific directories
resource "local_file" "in_directory" {
  filename             = "${path.module}/output/configs/app/settings.json"
  content              = jsonencode({ setting = "value" })
  directory_permission = "0755"
  file_permission      = "0644"
}
```

## Conclusion

The local_file resource is a versatile tool for generating any type of file content as part of your Terraform workflow. Whether you are creating JSON configuration files, shell scripts, Docker Compose files, or Ansible inventories, it integrates naturally with Terraform's variable system and output data. Remember to set appropriate file permissions, especially for scripts (0755) and configuration files (0644). For files containing secrets, use [local_sensitive_file](https://oneuptime.com/blog/post/2026-02-23-how-to-create-local-sensitive-files-with-terraform/view) instead, which restricts permissions and marks content as sensitive.
